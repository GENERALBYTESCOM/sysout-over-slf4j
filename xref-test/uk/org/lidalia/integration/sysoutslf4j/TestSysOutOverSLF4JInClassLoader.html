63  	@Test
64  	public void sysOutOverSLF4JWorksInsideAnotherClassLoader() throws Exception {
65  		callSendSystemOutAndErrToSLF4JInClassLoader(app1ClassLoader);
66  		
67  		ISysOutUser sysOutUser1 = newInstanceInClassLoader(ISysOutUser.class, app1ClassLoader, SysOutUser.class, new Class[]{});
68  		
69  		Thread.currentThread().setContextClassLoader(app1ClassLoader);
70  		sysOutUser1.useSysOut();
71  		
72  		List<?> list1 = getRootAppender(app1ClassLoader);
73  		assertEquals(1, list1.size());
74  		ILoggingEvent loggingEvent = CrossClassLoaderTestUtils.moveToCurrentClassLoader(ILoggingEvent.class, list1.get(0));
75  		Assert.assertExpectedLoggingEvent(loggingEvent, "Logged", Level.INFO, null, SysOutUser.class.getName());
76  	}

78  	private <E> E newInstanceInClassLoader(
79  			Class<E> classToReturn, ClassLoader classLoader, Class<? extends E> classToGetInstanceOf,
80  			Class<?>[] constructorArgTypes, Object... constructorArgs) throws Exception {
81  		Class<?> class1 = classLoader.loadClass(classToGetInstanceOf.getName());
82  		Object newInstance = Whitebox.invokeConstructor(class1, constructorArgTypes, constructorArgs);
83  		return CrossClassLoaderTestUtils.moveToCurrentClassLoader(classToReturn, newInstance);
84  	}

86  	static void resetSysOutUserAppender(ClassLoader classLoader) throws Exception {
87  		Class<?> clazz = classLoader.loadClass(TestSysOutOverSLF4JInClassLoader.class.getName());
88  		clazz.getDeclaredMethod("resetSysOutUserAppender").invoke(clazz);
89  	}
90  	
91  	public static void resetSysOutUserAppender() {
92  		LoggerContext loggerContext = (LoggerContext) LoggerFactory.getILoggerFactory();
93  		Logger sysOutUserLogger = loggerContext.getLogger(SysOutUser.class.getName());
94  		sysOutUserLogger.detachAndStopAllAppenders();
95  		ListAppender<ILoggingEvent> appender = new ListAppender<ILoggingEvent>();
96  		appender.setName("list");
97  		appender.setContext(loggerContext);
98  		appender.start();
99  		sysOutUserLogger.addAppender(appender);
100 		sysOutUserLogger.setLevel(Level.INFO);
101 	}

103 	static List<?> getRootAppender(ClassLoader classLoader) throws Exception {
104 		Class<?> clazz = classLoader.loadClass(TestSysOutOverSLF4JInClassLoader.class.getName());
105 		Object listAppender = clazz.getDeclaredMethod("getRootAppender").invoke(clazz);
106 		Class<?> listAppenderClass = classLoader.loadClass(ListAppender.class.getName());
107 		Field listField = listAppenderClass.getField("list");
108 		Object list = listField.get(listAppender);
109 		return CrossClassLoaderTestUtils.moveToCurrentClassLoader(List.class, list);
110 	}

112 	public static ListAppender<ILoggingEvent> getRootAppender() {
113 		LoggerContext LC = (LoggerContext) LoggerFactory.getILoggerFactory();
114 		Logger sysOutLogger = LC.getLogger(SysOutUser.class.getName());
115 		return (ListAppender<ILoggingEvent>) sysOutLogger.getAppender("list");
116 	}
