27  import static uk.org.lidalia.sysoutslf4j.context.ClassLoaderUtils.loadClass;
28  import static java.lang.Thread.currentThread;
29  
30  import java.util.concurrent.Callable;
31  
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  import uk.org.lidalia.sysoutslf4j.common.ExceptionUtils;
36  import uk.org.lidalia.sysoutslf4j.common.PrintStreamCoordinator;
37  import uk.org.lidalia.sysoutslf4j.common.ReflectionUtils;
38  import uk.org.lidalia.sysoutslf4j.system.PrintStreamCoordinatorImpl;

40  final class PrintStreamCoordinatorFactory {
41  	
42  	private static final String LINE_END = System.getProperty("line.separator");
43  	private static final Logger LOG = LoggerFactory.getLogger(SysOutOverSLF4J.class);
44  
45  	static PrintStreamCoordinator createPrintStreamCoordinator() {
46  		Class<?> candidateCoordinatorClass = getConfiguratorClassFromSLF4JPrintStreamClassLoader();
47  		if (candidateCoordinatorClass == null) {
48  			candidateCoordinatorClass = getConfiguratorClassFromSystemClassLoader();
49  		}
50  		if (candidateCoordinatorClass == null) {
51  			candidateCoordinatorClass = getConfiguratorClassFromCurrentClassLoader();
52  		}
53  		checkCoordinator(candidateCoordinatorClass);
54  		return makeCoordinator(candidateCoordinatorClass);
55  	}
56  
57  	private static PrintStreamCoordinator makeCoordinator(final Class<?> coordinatorClass) {
58  		return ExceptionUtils.doUnchecked(new Callable<PrintStreamCoordinator>() {
59  			@Override
60  			public PrintStreamCoordinator call() throws InstantiationException, IllegalAccessException {
61  				final Object coordinator = coordinatorClass.newInstance();
62  				return ReflectionUtils.wrap(coordinator, PrintStreamCoordinator.class);
63  			}
64  		});
65  	}
66  
67  	private static Class<?> getConfiguratorClassFromSLF4JPrintStreamClassLoader() {
68  		final Class<?> configuratorClass;
69  		if (SysOutOverSLF4J.systemOutputsAreSLF4JPrintStreams()) {
70  			final ClassLoader classLoader = System.out.getClass().getClassLoader(); // NOPMD deliberately using this classloader
71  			configuratorClass = loadClass(classLoader, PrintStreamCoordinatorImpl.class);
72  		} else {
73  			configuratorClass = null;
74  		}
75  		return configuratorClass;
76  	}
77  
78  	private static Class<?> getConfiguratorClassFromSystemClassLoader() {
79  		Class<?> configuratorClass = null;
80  		try {
81  			configuratorClass = ClassLoader.getSystemClassLoader().loadClass(PrintStreamCoordinatorImpl.class.getName());
82  		} catch (Exception e) {
83  			LOG.debug("failed to load [" + PrintStreamCoordinatorImpl.class + "] from system class loader due to " + e);
84  		}
85  		return configuratorClass;
86  	}
87  	
88  	private static void checkCoordinator(final Class<?> candidateCoordinatorClass) {
89  		final ClassLoader contextClassLoader = currentThread().getContextClassLoader();
90  		final boolean usingSystemClassLoader = ClassLoader.getSystemClassLoader() == contextClassLoader;
91  		if (!usingSystemClassLoader && candidateCoordinatorClass.getClassLoader() == contextClassLoader) {
92  			reportFailureToAvoidClassLoaderLeak();
93  		}
94  	}
95  
96  	private static void reportFailureToAvoidClassLoaderLeak() {
97  		LOG.warn("Unfortunately it is not possible to set up Sysout over SLF4J on this system without introducing " +
98  				"a class loader memory leak." + LINE_END +
99  				"If you never need to discard the current class loader [" + currentThread().getContextClassLoader() + "] " +
100 				"this will not be a problem and you can suppress this warning." + LINE_END +
101 				"In the worst case discarding the current class loader may cause all subsequent attempts to print to " +
102 				"System.out or err to throw an exception.");
103 	}
104 
105 	private static Class<PrintStreamCoordinatorImpl> getConfiguratorClassFromCurrentClassLoader() {
106 		return PrintStreamCoordinatorImpl.class;
107 	}
108 	
109 	private PrintStreamCoordinatorFactory() {
110 		throw new UnsupportedOperationException("Not instantiable");
111 	}
112 }
