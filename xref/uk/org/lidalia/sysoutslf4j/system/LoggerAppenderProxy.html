25  package uk.org.lidalia.sysoutslf4j.system;
26  
27  import java.lang.reflect.Method;
28  
29  import uk.org.lidalia.sysoutslf4j.common.LoggerAppender;
30  import uk.org.lidalia.sysoutslf4j.common.ReflectionUtils;
31  
32  final class LoggerAppenderProxy implements LoggerAppender {
33  	
34  	private final Object targetLoggerAppender;
35  	private final Method appendMethod;
36  	private final Method appendAndLogMethod;
37  
38  	private LoggerAppenderProxy(final Object targetLoggerAppender) {
39  		super();
40  		try {
41  			final Class<?> loggerAppenderClass = targetLoggerAppender.getClass();
42  			this.targetLoggerAppender = targetLoggerAppender;
43  			this.appendMethod = loggerAppenderClass.getDeclaredMethod("append", String.class);
44  			this.appendAndLogMethod =
45  				loggerAppenderClass.getDeclaredMethod("appendAndLog", String.class, String.class, boolean.class);
46  		} catch (NoSuchMethodException e) {
47  			throw new IllegalArgumentException(
48  					"Must only be instantiated with a LoggerAppender instance, got a " + targetLoggerAppender.getClass(), e);
49  		}
50  	}
51  
52  	public void append(final String message) {
53  		ReflectionUtils.invokeMethod(appendMethod, targetLoggerAppender, message);
54  	}
55  
56  	public void appendAndLog(final String message, final String className, final boolean isStackTrace) {
57  		ReflectionUtils.invokeMethod(appendAndLogMethod, targetLoggerAppender, message, className, isStackTrace);
58  	}
59  	
60  	static LoggerAppender wrap(final Object targetLoggerAppender) {
61  		final LoggerAppender result;
62  		if (targetLoggerAppender instanceof LoggerAppender) {
63  			result = (LoggerAppender) targetLoggerAppender;
64  		} else {
65  			result = new LoggerAppenderProxy(targetLoggerAppender);
66  		}
67  		return result;
68  	}
69  }
